package Factorial;
import java.util.Scanner;
public class Factorial {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        System.out.print("Enter a number: ");
        int n = input.nextInt();

        long result = 1;
        for (int i = 1; i <= n; i++) {
            result = result * i;
        }

        System.out.println("Factorial of " + n + " is: " + result);
    }
}
