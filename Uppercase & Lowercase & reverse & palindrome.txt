package secondclass;
import java.util.*;
public class Secondclass
{
    public static void main(String[] args) 
      {
        Scanner input = new Scanner(System.in);
        System.out.println("Enter a string: ");
        String str = input.nextLine();
       //Convert to Uppercase to make it case-insensitive (optional)
       str = str.toUpperCase();
         System.out.println("Upperercase string : " + str);
        // Convert to lowercase to make it case-insensitive (optional)
        str = str.toLowerCase();
         System.out.println("lowercase string : " + str);
        // Reverse the string
        String reversed = "";
        for (int i = str.length() - 1; i >= 0; i--) {
            reversed = reversed + str.charAt(i);
        }
        System.out.println("reverse string : " + reversed);
        
       // Check if original and reversed strings are equal
       if (str == reversed) 
        {
          System.out.println("It is a palindrome.");
        }
      else {
            System.out.println("It is not a palindrome.");
           }
        
      }
}
